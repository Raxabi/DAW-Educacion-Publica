-- Usando ON
SELECT CLIENTES.NOMBRE,
PEDIDOS.ID_CLIENTE,
PEDIDOS.FECHA,
PEDIDOS.IMPORTE
FROM CLIENTES
INNER JOIN PEDIDOS
ON CLIENTES.ID_CLIENTE = PEDIDOS.ID_CLIENTE
WHERE PEDIDOS.ID_CLIENTE = CLIENTES.ID_CLIENTE;

-- Usando USING
SELECT
CLIENTES.NOMBRE,
PEDIDOS.FECHA,
PEDIDOS.IMPORTE
FROM CLIENTES
INNER JOIN PEDIDOS USING (ID_CLIENTE)
WHERE PEDIDOS.ID_CLIENTE = CLIENTES.ID_CLIENTE;
